CREATE DATABASE CONTATOS

CREATE TABLE contato(
id INT NOT NULL AUTO_INCREMENT ,
nome VARCHAR(120) NOT NULL,
logradouro VARCHAR(120),
numero INT,
bairro VARCHAR(120),
cidade VARCHAR(120),
estado VARCHAR(120),
telefone VARCHAR(120) NOT NULL,
email VARCHAR(120) NOT NULL,
sexo CHAR,
dtNasc DATE,
PRIMARY KEY(id)
);
